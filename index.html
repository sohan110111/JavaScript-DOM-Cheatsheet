1Ô∏è‚É£1Ô∏è‚É£ Drag & Drop

Drag Me
const box = document.getElementById('dragBox');
let offsetX, offsetY;
box.onmousedown = e => {
  offsetX = e.offsetX;
  offsetY = e.offsetY;
  document.onmousemove = e => {
    box.style.left = e.pageX - offsetX + 'px';
    box.style.top = e.pageY - offsetY + 'px';
  }
  document.onmouseup = () => document.onmousemove = null;
};
